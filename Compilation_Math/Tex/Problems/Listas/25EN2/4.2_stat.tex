\begin{problem}
    [\cite{OrdenTomas} \gad]
    Demuestra que $a^{\frac{p-1}{2}}=\left( \frac{a}{p} \right)$
    donde $\left( \frac{a}{p} \right)$ es el simbolo de Legendre, y es igual a 0
    si $p \mid a$, 1 si $a$ es residuo cuadratico $\pmod p$ y $-1$ si no.
    \label{25EN2_4_2}
\end{problem}
